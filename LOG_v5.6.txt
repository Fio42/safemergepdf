PYTHON RECETAS — todas sin servidor
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
MERGE:    PdfWriter().append(BytesIO(bytes)) x N -> write(output)
COMPRESS: compress_identical_objects(True,True) + img.replace(quality=N)
SPLIT ALL: por pagina: PdfWriter().add_page(page).write(out) -> array
SPLIT RANGE: PdfWriter() con paginas seleccionadas -> write(out)
JPG->PDF: Pillow Image.open -> img.save(buf, format='PDF') -> PdfWriter.append
ROTATE:   page.rotate(90|180|270)
PROTECT:  writer.encrypt(password, algorithm="AES-256")
UNLOCK:   reader.decrypt(password); writer.append(reader)
PDF->JPG: PDF.js + canvas.toDataURL('image/jpeg', 0.92) — NO Pyodide
